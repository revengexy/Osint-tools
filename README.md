1. clone the project:
   git clone https://github.com/revengexy/Osint-tools
   cd Osint-tools

2. virtual terminal active
   python -m venv venv
   # Linux/macOS :
   source venv/bin/activate
   # Windows :
   venv\Scripts\activate

3. required pips
   pip install -r requirements.txt

4. run
   python osint.py
